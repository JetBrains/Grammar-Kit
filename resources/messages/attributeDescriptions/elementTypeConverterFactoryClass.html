  {
    elementTypeClass="org.intellij.grammar.psi.BnfCompositeElementType"
    elementTypeHolderClass="org.intellij.grammar.psi.BnfTypes"
    elementTypePrefix="BNF_"
    syntaxElementTypeHolderClass="org.intellij.grammar.BnfSyntaxTypes"
    elementTypeConverterFactoryClass="org.intellij.grammar.BnfSyntaxElementTypeConverterFactory"
